y <- 12
fibonacciSeries <- rep(0,y) ##used rep() to define an array of y elements of value zero
fibonacciSeries[2] <- 1


if(y>2){ ##to start after 2 as first two entries of the series are already defined as 0 and 1
    for (i in 3:y) { ##the argument specified in the loop will iterate for every value in the specified range until the argument remains true
        fibonacciSeries[i] <- fibonacciSeries[i-1] + fibonacciSeries[i-2] ##this portion of the code will sum two subsequent numbers on the series and hence add on to the vector fibonacciSeries
    }
}
